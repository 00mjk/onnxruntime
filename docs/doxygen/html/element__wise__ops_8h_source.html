325 struct BroadcastIterator {
326  size_t AdvanceBy(size_t delta) {
327  size_t index = index_;
328 
329  index_ += deltas_[0] * delta;
330  counters_[0] += delta;
331  if (counters_[0] == counts_[0]) {
332  counters_[0] = 0;
333  for (size_t counterIndex = 1; counterIndex < counters_.size(); counterIndex++) {
334  index_ += deltas_[counterIndex];
335  if (++counters_[counterIndex] != counts_[counterIndex])
336  break;
337  counters_[counterIndex] = 0;
338  }
339  }
340  return index;
341  }

343  void Init(int64_t axis, int64_t largest) {
344  ONNXRUNTIME_ENFORCE(axis == 1 || axis == largest, "Attempting to broadcast an axis by a dimension other than 1. ", axis, " by ", largest);
345 
346  deltas_.push_back(axis > 1);
347  counts_.push_back(largest);
348  count_ *= axis;
349  }
350 
351  void Append(int64_t axis, int64_t largest) {
352  ONNXRUNTIME_ENFORCE(axis == 1 || axis == largest, "Attempting to broadcast an axis by a dimension other than 1. ", axis, " by ", largest);
353 
354  // If we're greater than 1, it doesn't matter what the other tensor does
355  if (axis > 1) {
356  if (deltas_.back() <= 0) // Were we broadcasting
357  StopBroadcasting();
358  } else { // We must be 1, at this point
359  if (deltas_.back() > 0)
360  StartBroadcasting();
361  }
362 
363  counts_.back() *= largest; // Just increase the last count
364  count_ *= axis;
365  }
366 
367  void StopBroadcasting() {
368  deltas_.push_back(count_);
369  counts_.push_back(1);
370  }
371 
372  void StartBroadcasting() {
373  deltas_.push_back(-count_);
374  counts_.push_back(1);
375  }
376 
377  std::vector<int64_t> counters_;
378  std::vector<ptrdiff_t> deltas_;
379  std::vector<int64_t> counts_;
380  size_t count_{1}; // Running total count of entries in tensor, used while building up the entries
381 
382  private:
383  size_t index_{};
384 };

386 struct Broadcaster {
387  Broadcaster(const std::vector<int64_t>& shape1, const std::vector<int64_t>& shape2) {
388  size_t dimension_count_max = std::max(shape1.size(), shape2.size());
389  size_t dimension_count_min = std::min(shape1.size(), shape2.size());
390  output_shape_.resize(dimension_count_max);
391 
392  auto iter1 = shape1.end();
393  auto iter2 = shape2.end();
394  auto output_shape = output_shape_.end();
395 
396  // Scalars are a special case, as it's always a broadcast
397  size_t index = 0;
398  if (dimension_count_min == 0) {
399  if (shape1.size() == 0) // Shape1 is a scalar
400  {
401  if (shape2.size() == 0) // Two scalars?
402  {
403  iterator1_.Init(1, 1);
404  iterator2_.Init(1, 1);
405  } else {
406  auto axis = *--iter2;
407  iterator1_.Init(1, axis);
408  iterator2_.Init(axis, axis);
409  *--output_shape = axis;
410  }
411  } else { // Shape2 is a scalar
412  auto axis = *--iter1;
413  iterator1_.Init(axis, axis);
414  iterator2_.Init(1, axis);
415  *--output_shape = axis;
416  }
417  index++; // Manually increment since we processed one axis
418  }
419 
420  for (; index < dimension_count_min; index++) {
421  auto axis1 = *--iter1;
422  auto axis2 = *--iter2;
423 
424  auto largest = std::max(axis1, axis2);
425  *--output_shape = largest;
426 
427  if (largest == 1 && index + 1 < dimension_count_min) // Nothing to do in this case
428  continue;
429 
430  iterator1_.Init(axis1, largest);
431  iterator2_.Init(axis2, largest);
432  index++; // Manually increment since we processed one axis
433  break;
434  }
435 
436  for (; index < dimension_count_min; index++) {
437  auto axis1 = *--iter1;
438  auto axis2 = *--iter2;
439 
440  auto largest = std::max(axis1, axis2);
441  *--output_shape = largest;
442 
443  if (largest == 1) // Nothing to do in this case
444  continue;
445 
446  iterator1_.Append(axis1, largest);
447  iterator2_.Append(axis2, largest);
448  }
449 
450  // If one shape is bigger than another we need to broadcast the smaller onto the bigger from this point on
451  for (; index < dimension_count_max; index++) {
452  if (dimension_count_max == shape2.size()) {
453  auto axis = *--iter2;
454  iterator1_.Append(1, axis);
455  iterator2_.Append(axis, axis);
456  *--output_shape = axis;
457  } else {
458  auto axis = *--iter1;
459  iterator1_.Append(axis, axis);
460  iterator2_.Append(1, axis);
461  *--output_shape = axis;
462  }
463  }
464 
465  // Allocate the counters
466  iterator1_.counters_.resize(iterator1_.counts_.size(), 0);
467  iterator2_.counters_.resize(iterator2_.counts_.size(), 0);
468  }
469 
470  size_t GetSpanSize() const { return std::min(iterator1_.counts_.front(), iterator2_.counts_.front()); }
471 
472  BroadcastIterator iterator1_, iterator2_;
473  std::vector<int64_t> output_shape_;
474 };

561 // Broadcast loop for when using eigen, functions are in this form:
562 // Input0Scalar: [](EigenVectorMap<T> output, T input0, ConstEigenVectorMap<T> input1)
563 // Input1Scalar: [](EigenVectorMap<T> output, ConstEigenVectorMap<T> input0, T input1)
564 // General : [](EigenVectorMap<T> output, ConstEigenVectorMap<T> input0, ConstEigenVectorMap<T> input1)
565 template <typename TBroadcaster, typename Output, typename Input0Scalar, typename Input1Scalar, typename General>
566 void BroadcastLoop(TBroadcaster& bc, Output& output, Input0Scalar input0scalar, Input1Scalar input1scalar, General general) {
567  if (bc.IsInput0Scalar()) {
568  while (output)
569  input0scalar(output.NextEigenOutput(), bc.NextScalar0(), bc.NextEigen1());
570  } else if (bc.IsInput1Scalar()) {
571  while (output)
572  input1scalar(output.NextEigenOutput(), bc.NextEigen0(), bc.NextScalar1());
573  } else {
574  while (output)
575  general(output.NextEigenOutput(), bc.NextEigen0(), bc.NextEigen1());
576  }
577 }

588 template <typename TInput, typename TOutput, typename Input0Scalar, typename Input1Scalar, typename General>
589 Status BroadcastVariadic(const Node& node, OpKernelContext& context, Input0Scalar input0scalar, Input1Scalar input1scalar, General general) {
590  auto input_count = node.InputArgCount().front();
591  ONNXRUNTIME_ENFORCE(input_count >= 1, "Must have 1 or more inputs");
592 
593  // One item is trivial, just copy across and exit
594  if (input_count == 1) {
595  EigenMap<TOutput>(*context.Output(0, context.Input<Tensor>(0)->Shape())) = EigenMap<TInput>(*context.Input<Tensor>(0));
596  return Status::OK();
597  }
598 
599  std::unique_ptr<Tensor> tempInput;
600  std::unique_ptr<Tensor> tempOutput;
601 
602  TensorAllocator<TOutput> tensorAllocator(context);
603 
604  // For more than 2 tensors, we sum the first two into a temporary tensor, then sum the next with the temporary tensor
605  for (int i = 0; i < input_count - 1; i++) {
606  auto& tensor0 = tempInput ? *tempInput : *context.Input<Tensor>(0);
607  auto& tensor1 = *context.Input<Tensor>(i + 1);
608 
609  TBroadcaster<TInput> bc(tensor0, tensor1);
610 
611  // Create a temporary output for all but the last iteration, which goes to the real output
612  Tensor* p_output{};
613  if (i == input_count - 2)
614  p_output = context.Output(0, bc.GetOutputShape());
615  else {
616  tempOutput = tensorAllocator.Allocate(bc.GetOutputShape());
617  p_output = tempOutput.get();
618  }
619 
620  TBroadcastOutput<TOutput> output(bc.GetSpanSize(), *p_output);
621 
622  BroadcastLoop(bc, output, input0scalar, input1scalar, general);
623 
624  tempInput = std::move(tempOutput);
625  }
626  return Status::OK();
627 }
