160  struct EdgeEndCompare {
161  bool operator()(const EdgeEnd& lhs, const EdgeEnd& rhs) const {
162  if (lhs.GetNode().Index() == rhs.GetNode().Index()) {
163  auto lhs_arg = lhs.GetNodeArg();
164  auto rhs_arg = rhs.GetNodeArg();
165  std::string lhs_arg_name = lhs_arg == nullptr ? "" : lhs_arg->Name();
166  std::string rhs_arg_name = rhs_arg == nullptr ? "" : rhs_arg->Name();
167  return lhs_arg_name.compare(rhs_arg_name) < 0;
168  }
169  return lhs.GetNode().Index() < rhs.GetNode().Index();
170  }
171  };
